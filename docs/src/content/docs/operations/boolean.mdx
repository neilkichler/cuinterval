## less

<FunctionDeclaration
  sourceUrl="include/cuinterval/arithmetic/basic.cuh#L579" 
  nvidiaUrl="group__CUDA__MATH__DOUBLE.html#_CPPv43sind"
>
  bool less(<IntervalRef /> a, <IntervalRef /> b)
</FunctionDeclaration>

<FunctionDetails id="less">
  <FunctionBrief error="0" slot="brief">
    Returns true if $$a \leq b$$.
  </FunctionBrief>
  
  <FunctionSignature>
    $$\leq: \mathbb{IR}\times\mathbb{IR} \rightarrow \mathbb{B}$$
  </FunctionSignature>
  For $a, b\neq \emptyset$:

<Centered>
    $\underline{a} \leq \underline{b} \land \bar{a} \leq \bar{b}$.
</Centered>

Otherwise:
<div style="margin-left: auto; margin-right: auto; width: 35%">
    | $a = \emptyset$ | $b = \emptyset$ | $a \leq b$ |
    |:---------------:|:---------------:|:----------:|
    |      $0$        |      $1$        |     $0$    |
    |      $1$        |      $0$        |     $0$    |
    |      $1$        |      $1$        |     $1$    |
</div>

</FunctionDetails>
