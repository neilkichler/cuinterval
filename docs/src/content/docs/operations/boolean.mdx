## equal

<FunctionDeclaration
  sourceUrl="include/cuinterval/arithmetic/basic.cuh#L633" 
  nvidiaUrl="group__CUDA__MATH__DOUBLE.html#equald"
>
  bool equal(<IntervalRef /> a, <IntervalRef /> b)
</FunctionDeclaration>

<FunctionDetails id="equal">
  <FunctionBrief error="0" slot="brief">
    Returns true if intervals $$a$$ and $$b$$ are equal.
  </FunctionBrief>
  
  <FunctionSignature>
    $$=: \mathbb{IR}\times\mathbb{IR} \rightarrow \mathbb{B}$$
  </FunctionSignature>
  For $a, b\neq \emptyset$:

<Centered>
    $\underline{a} = \underline{b} \land \bar{a} = \bar{b}$.
</Centered>

Otherwise:
<div style="margin-left: auto; margin-right: auto; width: 35%">
    | $a = \emptyset$ | $b = \emptyset$ | $a = b$ |
    |:---------------:|:---------------:|:-------:|
    |      $0$        |      $1$        |   $0$   |
    |      $1$        |      $0$        |   $0$   |
    |      $1$        |      $1$        |   $1$   |
</div>

</FunctionDetails>
